## Examples

Many different `ShopLayout1` usage in this templates

```jsx
import ShopLayout1 from 'components/layouts/ShopLayout1'

// example 1 - /market-1
const Example1 = () => {
  return <ShopLayout1>{children}</ShopLayout1>
}

// example 2 - /fashion-shop-1
const Example2 = () => {
  return <ShopLayout1 showTopbar={false}>{children}</ShopLayout1>
}

// example 3 - /grocery3
const Example3 = () => {
  return <ShopLayout1 showNavbar={false}>{children}</ShopLayout1>
}

// example 4 - /market-2
const Example4 = () => {
  return <ShopLayout1 topbarBgColor="red">{children}</ShopLayout1>
}
```

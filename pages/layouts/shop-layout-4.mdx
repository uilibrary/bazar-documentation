## Examples

```jsx {1} copy showLineNumbers
import ShopLayout4 from 'components/layouts/shop-layout-4'

export default function Layout({ children }) {
  const navigation = [
    {
      name: 'Grocery',
      icon: 'Grocery',
      path: '/furniture-3/products/grocery',
      children: [
        {
          name: 'Fruits & Vegetables',
          path: '/furniture-3/products/fruits-vegetables',
          children: [
            { name: 'Fruits', path: '/furniture-3/products/fruits' },
            { name: 'Vegetables', path: '/furniture-3/products/vegetables' },
          ],
        },
        {
          name: 'Diabetic Food',
          path: '/furniture-3/products/diabetic-food',
        },
      ],
    },
  ]

  return <ShopLayout4 navigation={navigation}>{children}</ShopLayout4>
}
```

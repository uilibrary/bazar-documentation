## Examples

```jsx
import ShopLayout2 from 'components/layouts/ShopLayout2'

// example 1 - /healthbeauty-shop
const Example1 = () => {
  return <ShopLayout2>{children}</ShopLayout2>
}

// example 2 - /grocery2
const Example2 = () => {
  return <ShopLayout2 showTopbar={false}>{children}</ShopLayout2>
}

// example 3 - /grocery1
const Example3 = () => {
  return (
    <ShopLayout2 showNavbar={false} showTopbar={false}>
      {children}
    </ShopLayout2>
  )
}
```
